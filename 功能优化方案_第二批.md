## 需求1：会议论文格式优化

### 1.1 现状问题

**parser 解析错误**（`src/lib/parser.ts:204-226`）：
```
输入：[1]栾成显.赋役黄册与明代等级身份[C]//中国明史学会,...学会.论文集名.天津古籍出版社,2005:12-23.
```
当前 `placeMatch = /([^,.，。]+)[,，]\s*\d{4}/` 把"天津古籍出版社"当成了**出版地**，出版社字段完全没提取。

**formatter 输出错误**（`src/lib/formatters/lsyj.ts:243-263`）：

```

```
缺少论文集名、出版地、出版社。

## 需求2：英文文献解析与格式优化

### 2.1 现状问题

1. **parser 只支持 APA**（`parser.ts:377-461`）：有 `(Year).` 才走 APA 分支，MLA 格式（`wilensky, norman m. "Title." Journal, vol. 13, no. 1, 1972, pp. 177-87.`）会走降级分支，作者/标题提取混乱。
2. **英文期刊 formatter 缺引号和斜体提示**（`lsyj.ts:191-200`）：标题没加英文引号，期刊名没斜体标记。
3. **粘贴提示文案需改**（`ConvertPage.tsx:335-340`）。

### 2.2 改法

#### 2.2.1 新增 MLA 解析器（`src/lib/parser.ts`）

在 `parseEnglishAPA` 之前加 MLA 检测：

```ts
export function parseEnglishCitation(text: string): Partial<Citation> {
  // 先尝试 MLA：特征是标题用双引号 "..." 包裹
  if (/[""]/.test(text)) {
    const mla = parseMLA(text)
    if (mla.title) return mla
  }
  // 再尝试 APA
  return parseEnglishAPA(text)
}

function parseMLA(text: string): Partial<Citation> {
  const result: Partial<Citation> = {}

  // 作者：第一个双引号前的内容
  const quoteIdx = text.indexOf('"')
  if (quoteIdx < 0) return result
  const authorsPart = text.slice(0, quoteIdx).trim().replace(/[,.]\s*$/, '')
  result.authors = parseMLAAuthors(authorsPart)

  // 标题：双引号内的内容
  const titleMatch = /"([^"]+)"/.exec(text)
  if (titleMatch) result.title = titleMatch[1].trim()

  // 双引号后的内容：期刊名, vol. X, no. Y, YEAR, pp. Z. Database, URL. Accessed DATE.
  const afterTitle = text.slice(titleMatch.index + titleMatch[0].length).trim().replace(/^[.,]\s*/, '')

  // 期刊名：第一个逗号前（或句号前）
  const journalMatch = /^([^,，.]+)[,，.]\s*/.exec(afterTitle)
  if (journalMatch) {
    result.journalName = journalMatch[1].trim()
    var rest = afterTitle.slice(journalMatch[0].length)
  } else {
    var rest = afterTitle
  }

  // vol. X
  const volMatch = /vol\.?\s*(\d+)/i.exec(rest)
  if (volMatch) result.volumeNumber = volMatch[1]

  // no. Y
  const noMatch = /no\.?\s*(\d+)/i.exec(rest)
  if (noMatch) result.issue = noMatch[1]

  // 年份
  const yearMatch = /(\d{4})/.exec(rest)
  if (yearMatch) result.publishYear = yearMatch[1]

  // 页码 pp. Z 或 p. Z
  const pageMatch = /p+p?\.\s*([\d–—-]+)/i.exec(rest)
  if (pageMatch) result.pages = pageMatch[1]

  // URL
  const urlMatch = /(https?:\/\/[^\s]+)/.exec(rest)
  if (urlMatch) result.url = urlMatch[1]

  // 访问日期 Accessed DATE
  const accessMatch = /accessed\s+([\w\s,]+)/i.exec(rest)
  if (accessMatch) result.accessDate = accessMatch[1].trim()

  result.type = 'journal'
  return result
}

function parseMLAAuthors(authorsPart: string): { name: string }[] {
  if (!authorsPart.trim()) return []
  // MLA: "wilensky, norman m." → 规范化为 "Wilensky, Norman M."
  // 多作者用 "; " 或 ", and " 分隔
  const segments = authorsPart.split(/;\s*|,\s*and\s+/i).map(s => s.trim()).filter(Boolean)
  return segments.map(raw => {
    // 首字母大写
    const parts = raw.split(/,\s*/).map(p =>
      p.trim().split(/\s+/).map(w =>
        w.charAt(0).toUpperCase() + w.slice(1).toLowerCase()
      ).join(' ')
    )
    // parts[0]=姓, parts[1]=名（如果有）
    return { name: parts.length >= 2 ? `${parts[0]}, ${parts[1]}` : parts[0] }
  })
}
```

```ts
import { IconRepeat } from '../components/Icons'

// 生成再次引证版本（仅 LSYJ + book）
const generateReCitation = (): string | null => {
  if (targetFormat !== 'lsyj' || citation.type !== 'book') return null
  const auth = citation.authors.map(a => {
    if (citation.language === 'en') {
      const suffix = a.role && a.role !== 'author' ? `, ${a.role}` : ''
      return a.name + suffix
    }
    if (!a.role || a.role === '著') return a.name
    return a.name + a.role
  }).join(citation.language === 'en' ? ', ' : '、')
  const title = citation.language === 'en'
    ? `*${citation.title}*`
    : processBookTitleMarks(`《${citation.title}》`)
  const parts = [auth ? auth + '：' : '', title]
  if (citation.pages) {
    parts.push(citation.language === 'en' ? `p.${citation.pages}` : `第${citation.pages}页`)
  }
  return parts.join('') + (citation.language === 'en' ? '.' : '。')
}

const handleReCite = async () => {
  const reCited = generateReCitation()
  if (!reCited) return
  const ok = await copyToClipboard(reCited)
  if (ok) showToast('已复制再次引证版本')
}
```

#### 6.2.2 formatter 修改（`src/lib/formatters/lsyj.ts:190-230`）

修改中文期刊分支：

```ts
case 'journal': {
  if (isEn) { ... } // 英文逻辑不变

  // 中文期刊
  let journalPart = ''
  if (c.journalName) {
    const match = c.journalName.match(/^(.+?)\(([^()]+)\)$/) || c.journalName.match(/^(.+?)（([^()]+)）$/)
    if (match) {
      journalPart = processBookTitleMarks(`《${match[1].trim()}》`) + `（${match[2]}）`
    } else {
      journalPart = processBookTitleMarks(`《${c.journalName}》`)
    }
  }
  const auth = authorStr(c)
  let result = ''
  if (auth) result = auth + '：'
  result += processBookTitleMarks(`《${c.title}》`)
  result += '，'
  result += journalPart

  if (c.volumeNumber && c.issue) {
    // 有卷号：第X卷第Y号，YYYY年（规范用"号"）
    result += `第${c.volumeNumber}卷第${removeLeadingZero(c.issue)}号`
    if (c.publishYear) result += `，${c.publishYear}年`
  } else {
    // 无卷号：YYYY年第N期
    if (c.publishYear) result += `${c.publishYear}年`
    if (c.issue) result += `第${removeLeadingZero(c.issue)}期`
  }
  // 页码默认不输出（由结果区"显示页码"按钮控制）
  return result + '。'
}
```

```ts
const [showPages, setShowPages] = useState(false)
```
